local player = game.Players.LocalPlayer
local character = player.Character or player.CharacterAdded:Wait()
local hrp = character:WaitForChild("HumanoidRootPart")

local autoFarm = true
local attackDistance = 3

-- CHỈNH THEO GAME
local mobFolder = workspace:WaitForChild("Mobs")
local leaderstats = player:WaitForChild("leaderstats")
local levelValue = leaderstats:WaitForChild("Level")

-- Tìm quái gần nhất
local function getMob()
    for _, mob in pairs(mobFolder:GetChildren()) do
        if mob:FindFirstChild("Humanoid")
        and mob.Humanoid.Health > 0
        and mob:FindFirstChild("HumanoidRootPart") then
            return mob
        end
    end
end

-- Auto Farm Loop
task.spawn(function()
    while autoFarm do
        local mob = getMob()

        if mob then
            -- Teleport
            hrp.CFrame = mob.HumanoidRootPart.CFrame * CFrame.new(0, 0, attackDistance)

            -- Attack bằng Tool
            for _, tool in pairs(character:GetChildren()) do
                if tool:IsA("Tool") then
                    tool:Activate()
                end
            end
        end

        task.wait(0.25)
    end
end)

-- Log level lên
levelValue:GetPropertyChangedSignal("Value"):Connect(function()
    print("Level mới:", levelValue.Value)
end)
